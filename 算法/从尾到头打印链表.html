		1.先反转链表（新建链表），然后打印:
		ListNode* reverse(ListNode* pHead){
			ListNode* pRevrerseHead = NULL;
			ListNode* pPrev = NULL;
			ListNode* pNode = pHead;
			while(pNode != NULL){
				ListNode* pNext = pNode->next;
				if(pNext == NULL){
					pRevrerseHead=pNext;
				}
				pNode->next=pPrev;
				pPrev=pNode;
				pNode=pNext;
			}
			return pRevrerseHead;
		}
