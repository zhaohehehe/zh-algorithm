1.将字符串中的空格替换成%20

			O(n^2)时间复杂度就是字符串移动，O(n)时间复杂度需要依靠指针；
			（1）后续遍历（移动指针不会覆盖）；
			（2）p1指向字符串的结尾；计算出所有空格的数量，依据空格数量计算出指针p2的位置(字符串的长度加上空格的长度)；
		采用JAVA语言模拟算法为：
		public static char[] find(char[] chs) {
		int spaceCount = 0;
		if (chs == null || chs.length <= 0) {
			return chs;
		}
		// 计算出空格的数量
		for (int i = 0; i < chs.length; i++) {
			if (chs[i] == ' ') {
				spaceCount++;
			}
		}
		char[] newChs = new char[chs.length + spaceCount * 2];
		int p1 = chs.length - 1;
		int p2 = newChs.length - 1;
		while (p1 >= 0 && p2 >= 0 && p1 <= p2) {
			if (chs[p1] != ' ') {
				newChs[p2--] = chs[p1];
			} else {
				newChs[p2--] = '0';
				newChs[p2--] = '2';
				newChs[p2--] = '%';
			}
			p1--;
		}
		return newChs;

	}

	public static void main(String[] args) {
		char[] chs = new char[] { 'w', 'e', ' ', 'a', 'r', 'e', ' ', 'o', 'n', 'e' };
		System.out.println(Test.find(chs));
	}
